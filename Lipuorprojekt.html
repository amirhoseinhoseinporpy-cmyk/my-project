    Dense(1, activation='sigmoid')  # خروجی باینری
])

    loss='binary_crossentropy',
    optimizer=Adam(),
    metrics=['accuracy', tf.keras.metrics.Precision(name='precision'), tf.keras.metrics.Recall(name='recall')]
)

# آموزش مدل
history=model.fit(

    validation_split=0.2,

def show_feature_statistics(df):

        stats = df[col].describe()

# سیستم ریکامندر بر اساس یک ویژگی
def recommend_by_feature(df, key, target, top_n=5):
    feature_range = df[key].max() - df[key].min()
    if feature_range == 0:
        df['score'] = 0
    else:
        df['score'] = 1 - (abs(df[key] - target) / feature_range)
    similar = df.copy().sort_values('score', ascending=False).head(top_n)
    return similar

def recommend_by_name(drink_name, df, similarity_matrix, top_n=5):
    drink_idx = df[df['wine_nigger'] == drink_name].index[0]
    similarity_scores = similarity_matrix[drink_idx]
    similar_indices = similarity_scores.argsort()[::-1][1:top_n+1]
    similar = df.iloc[similar_indices].copy()
    similar['score'] = similarity_scores[similar_indices]
    return similar

# چاپ خروجی
def display_recommended_drinks(drink_name, recommended_drinks):
    print(f"\nنتایج مشابه برای: {drink_name}")
    print("-" * 90)
    print("شماره | نام مشروب           | الکل (%) | pH     | sulphates | سایت فروشنده         | امتیاز شباهت")
    print("-" * 90)
    for idx, row in recommended_drinks.iterrows():
        print(f"{idx:<6} | {row['wine_nigger']:<20} | {row['alcohol']:<9.1f} | {row['pH']:<6.2f} | {row['sulphates']:<9.2f} | {row['ip']:<20} | {row['score']:.3f}")

# آماده‌سازی ماتریس شباهت برای مقایسه بر اساس اسم
X = df_final[['alcohol', 'pH', 'sulphates']].values
scaler = MinMaxScaler()
X_scaled = scaler.fit_transform(X)
features_with_prediction = np.hstack([X_scaled, df_final[['predicted_quality_prob']].values])
similarity_matrix = cosine_similarity(features_with_prediction)

while True:
