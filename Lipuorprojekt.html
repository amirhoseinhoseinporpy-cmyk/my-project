#   
def show_feature_statistics(df):
    print("\n---     ---\n")
    for col in ['alcohol', 'pH', 'sulphates']:
        stats = df[col].describe()
        print(f"{col}:\n  min: {stats['min']:.2f} | max: {stats['max']:.2f} | mean: {stats['mean']:.2f} | std: {stats['std']:.2f}\n")

#      
def recommend_by_feature(df, key, target, top_n=5):
    feature_range = df[key].max() - df[key].min()
    if feature_range == 0:
        df['score'] = 0
    else:
        df['score'] = 1 - (abs(df[key] - target) / feature_range)
    similar = df.copy().sort_values('score', ascending=False).head(top_n)
    return similar

#         
def recommend_by_name(drink_name, df, similarity_matrix, top_n=5):
    drink_idx = df[df['wine_nigger'] == drink_name].index[0]
    similarity_scores = similarity_matrix[drink_idx]
    similar_indices = similarity_scores.argsort()[::-1][1:top_n+1]
    similar = df.iloc[similar_indices].copy()
    similar['score'] = similarity_scores[similar_indices]
    return similar
